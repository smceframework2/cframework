###Install

##Ubuntu

$ cd build

$ sudo apt-get install php5-dev libpcre3-dev gcc make

$ ./install 

php.ini set.

extension=smce.so

$ service apache2 restart



##CentOS/RedHat/Fedora

$ cd build

$ sudo yum install php-devel pcre-devel gcc make

$ ./install 

php.ini set.

extension=smce.so

$ service apache2 restart


##Suse

$ cd build

$ sudo yast -i gcc make autoconf2.13 php5-devel php5-pear

$ ./install 

php.ini set.

extension=smce.so

$ service apache2 restart
